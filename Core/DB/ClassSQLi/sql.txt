// Создание таблицы для логов в админке

id автоинкримент
user_id - ид пользователя
user_name - Имя пользователя
log_text - Текст события
event_type - Тип события ( 1 - Авторизация / Выход , 2 - создание/удаление/изменение )
event_date - Дата и время события

CREATE TABLE admin_panel_logs ( id INTEGER PRIMARY KEY AUTOINCREMENT,user_id INTEGER, user_name VARCHAR(60), log_text TEXT, event_type INTEGER DEFAULT 2,event_date DATETIME )
